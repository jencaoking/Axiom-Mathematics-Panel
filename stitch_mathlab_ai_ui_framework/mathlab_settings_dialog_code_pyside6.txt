import sys
from PySide6.QtCore import Qt, Signal, QPoint, QSize
from PySide6.QtGui import QColor, QFont, QKeySequence
from PySide6.QtWidgets import (
    QApplication, QDialog, QVBoxLayout, QHBoxLayout, QTabWidget, 
    QWidget, QFormLayout, QComboBox, QPushButton, QFontComboBox, 
    QSpinBox, QDoubleSpinBox, QCheckBox, QSlider, QTableWidget, 
    QTableWidgetItem, QHeaderView, QDialogButtonBox, QLabel, 
    QGridLayout, QMessageBox, QFrame, QScrollArea
)

class SettingsDialog(QDialog):
    """
    MathLab Settings Dialog
    A modular preference window for customizing application appearance and behavior.
    """
    
    # Signals for real-time updates
    theme_changed = Signal(str)
    accent_color_changed = Signal(str)
    font_changed = Signal(str, int)
    language_changed = Signal(str)
    graphics_settings_changed = Signal(dict)
    console_settings_changed = Signal(dict)
    shortcuts_changed = Signal(dict)
    advanced_settings_changed = Signal(dict)

    def __init__(self, parent=None, initial_settings=None):
        super().__init__(parent)
        self.setWindowTitle("MathLab Settings")
        self.setMinimumSize(700, 500)
        
        # Default settings if none provided
        self.settings = initial_settings or self._get_default_values()
        
        self._init_ui()
        self._load_settings()

    def _get_default_values(self):
        return {
            "theme": "Light",
            "accent": "#2563EB",
            "ui_font": "Inter",
            "ui_font_size": 10,
            "canvas_bg": "Grid",
            "language": "English",
            "line_width": 1.5,
            "point_size": 4,
            "aa_enabled": True,
            "anim_speed": 50,
            "snap_enabled": True,
            "console_font": "Fira Code",
            "console_font_size": 11,
            "console_history": 1000,
            "shortcuts": {
                "New Project": "Ctrl+N",
                "Save": "Ctrl+S",
                "Undo": "Ctrl+Z",
                "Redo": "Ctrl+Shift+Z",
                "Clear Canvas": "Ctrl+Del",
                "Select Mode": "Esc",
                "Execute Code": "Shift+Enter"
            }
        }

    def _init_ui(self):
        layout = QVBoxLayout(self)
        layout.setContentsMargins(0, 0, 0, 0)
        layout.setSpacing(0)

        # Tab Widget
        self.tabs = QTabWidget()
        self.tabs.setTabPosition(QTabWidget.West) # Sidebar style tabs
        self.tabs.setStyleSheet("""
            QTabWidget::pane { border: none; background: white; }
            QTabBar::tab { 
                padding: 12px 20px; 
                text-align: left; 
                border-bottom: 1px solid #eee;
                min-width: 150px;
            }
            QTabBar::tab:selected { background: #f8f9ff; border-left: 4px solid #2563EB; font-weight: bold; }
        """)

        # Add Tabs
        self.tabs.addTab(self._create_appearance_tab(), "Appearance")
        self.tabs.addTab(self._create_graphics_tab(), "Graphics")
        self.tabs.addTab(self._create_console_tab(), "Console")
        self.tabs.addTab(self._create_shortcuts_tab(), "Shortcuts")
        self.tabs.addTab(self._create_advanced_tab(), "Advanced")

        layout.addWidget(self.tabs)

        # Buttons
        self.button_box = QDialogButtonBox(
            QDialogButtonBox.Ok | QDialogButtonBox.Cancel | QDialogButtonBox.Apply
        )
        self.button_box.button(QDialogButtonBox.Ok).setText("OK")
        self.button_box.clicked.connect(self._on_button_clicked)
        
        button_container = QWidget()
        button_container.setStyleSheet("background: #fcfcfc; border-top: 1px solid #ddd;")
        btn_layout = QHBoxLayout(button_container)
        btn_layout.addWidget(self.button_box)
        layout.addWidget(button_container)

    def _create_appearance_tab(self):
        page = QWidget()
        layout = QFormLayout(page)
        layout.setContentsMargins(30, 30, 30, 30)
        layout.setSpacing(15)

        # Theme
        self.theme_combo = QComboBox()
        self.theme_combo.addItems(["Light", "Dark", "Follow System"])
        layout.addRow("Theme", self.theme_combo)

        # Accent Color
        accent_layout = QHBoxLayout()
        colors = ["#2563EB", "#10B981", "#8B5CF6", "#F59E0B"]
        self.accent_group = []
        for color in colors:
            btn = QPushButton()
            btn.setFixedSize(24, 24)
            btn.setCheckable(True)
            btn.setStyleSheet(f"background-color: {color}; border-radius: 12px; border: 2px solid transparent;")
            btn.clicked.connect(lambda checked, c=color: self._set_accent_selection(c))
            accent_layout.addWidget(btn)
            self.accent_group.append((color, btn))
        accent_layout.addStretch()
        layout.addRow("Accent Color", accent_layout)

        # Font
        self.ui_font_combo = QFontComboBox()
        self.ui_font_size = QSpinBox()
        self.ui_font_size.setRange(8, 24)
        font_layout = QHBoxLayout()
        font_layout.addWidget(self.ui_font_combo)
        font_layout.addWidget(self.ui_font_size)
        layout.addRow("Interface Font", font_layout)

        # Canvas BG
        self.bg_combo = QComboBox()
        self.bg_combo.addItems(["Grid", "Blank", "Polar Grid"])
        layout.addRow("Canvas Background", self.bg_combo)

        # Language
        self.lang_combo = QComboBox()
        self.lang_combo.addItems(["English", "中文 (简体)"])
        layout.addRow("Language", self.lang_combo)
        
        return page

    def _create_graphics_tab(self):
        page = QWidget()
        layout = QFormLayout(page)
        layout.setContentsMargins(30, 30, 30, 30)
        layout.setSpacing(15)

        self.line_width_spin = QDoubleSpinBox()
        self.line_width_spin.setRange(0.1, 10.0)
        self.line_width_spin.setSingleStep(0.1)
        self.line_width_spin.setSuffix(" px")
        layout.addRow("Default Line Width", self.line_width_spin)

        self.point_size_spin = QSpinBox()
        self.point_size_spin.setRange(1, 20)
        self.point_size_spin.setSuffix(" px")
        layout.addRow("Default Point Size", self.point_size_spin)

        self.aa_check = QCheckBox("Enable Anti-aliasing")
        layout.addRow("Rendering", self.aa_check)

        self.speed_slider = QSlider(Qt.Horizontal)
        self.speed_slider.setRange(10, 200)
        layout.addRow("Animation Speed", self.speed_slider)

        return page

    def _create_console_tab(self):
        page = QWidget()
        layout = QFormLayout(page)
        layout.setContentsMargins(30, 30, 30, 30)

        self.console_font = QFontComboBox()
        self.console_font_size = QSpinBox()
        self.console_font_size.setRange(8, 30)
        
        font_layout = QHBoxLayout()
        font_layout.addWidget(self.console_font)
        font_layout.addWidget(self.console_font_size)
        layout.addRow("Console Font", font_layout)

        self.history_limit = QSpinBox()
        self.history_limit.setRange(100, 10000)
        layout.addRow("History Limit", self.history_limit)

        self.autocomplete_check = QCheckBox("Enable Auto-completion")
        layout.addRow("Intellisense", self.autocomplete_check)

        return page

    def _create_shortcuts_tab(self):
        page = QWidget()
        layout = QVBoxLayout(page)
        layout.setContentsMargins(20, 20, 20, 20)

        self.shortcut_table = QTableWidget(7, 2)
        self.shortcut_table.setHorizontalHeaderLabels(["Action", "Shortcut"])
        self.shortcut_table.horizontalHeader().setSectionResizeMode(QHeaderView.Stretch)
        layout.addWidget(self.shortcut_table)

        btn_restore = QPushButton("Restore Defaults")
        btn_restore.clicked.connect(self._restore_defaults)
        layout.addWidget(btn_restore, alignment=Qt.AlignRight)

        return page

    def _create_advanced_tab(self):
        page = QWidget()
        layout = QVBoxLayout(page)
        layout.setContentsMargins(30, 30, 30, 30)

        self.hw_accel = QCheckBox("Enable Hardware Acceleration (OpenGL)")
        layout.addWidget(self.hw_accel)

        auto_save_layout = QHBoxLayout()
        self.autosave_spin = QSpinBox()
        self.autosave_spin.setRange(0, 60)
        auto_save_layout.addWidget(QLabel("Auto-save interval:"))
        auto_save_layout.addWidget(self.autosave_spin)
        auto_save_layout.addWidget(QLabel("minutes (0 to disable)"))
        layout.addLayout(auto_save_layout)

        btn_reset = QPushButton("Reset All Settings...")
        btn_reset.setStyleSheet("color: #dc2626; border: 1px solid #fecaca; padding: 8px;")
        btn_reset.clicked.connect(self._reset_all)
        layout.addStretch()
        layout.addWidget(btn_reset)

        return page

    def _set_accent_selection(self, color_hex):
        for color, btn in self.accent_group:
            btn.setChecked(color == color_hex)
            btn.setStyleSheet(f"""
                background-color: {color}; 
                border-radius: 12px; 
                border: 2px solid {'#000' if color == color_hex else 'transparent'};
            """)
        self.settings["accent"] = color_hex

    def _load_settings(self):
        # Implementation of UI population from self.settings
        self.theme_combo.setCurrentText(self.settings.get("theme", "Light"))
        self._set_accent_selection(self.settings.get("accent", "#2563EB"))
        self.line_width_spin.setValue(self.settings.get("line_width", 1.5))
        self.point_size_spin.setValue(self.settings.get("point_size", 4))
        
        # Populate Shortcut Table
        for i, (action, keys) in enumerate(self.settings["shortcuts"].items()):
            self.shortcut_table.setItem(i, 0, QTableWidgetItem(action))
            self.shortcut_table.setItem(i, 1, QTableWidgetItem(keys))

    def _on_button_clicked(self, button):
        role = self.button_box.buttonRole(button)
        if role == QDialogButtonBox.ApplyRole:
            self._apply_settings()
        elif role == QDialogButtonBox.AcceptRole:
            self._apply_settings()
            self.accept()
        elif role == QDialogButtonBox.RejectRole:
            self.reject()

    def _apply_settings(self):
        """Emit signals with current UI values"""
        self.theme_changed.emit(self.theme_combo.currentText())
        self.accent_color_changed.emit(self.settings["accent"])
        self.font_changed.emit(self.ui_font_combo.currentFont().family(), self.ui_font_size.value())
        
        graphics = {
            "line_width": self.line_width_spin.value(),
            "point_size": self.point_size_spin.value(),
            "aa": self.aa_check.isChecked(),
            "speed": self.speed_slider.value()
        }
        self.graphics_settings_changed.emit(graphics)
        
        print(f"Settings applied: Theme={self.theme_combo.currentText()}, Accent={self.settings['accent']}")

    def _restore_defaults(self):
        self.settings["shortcuts"] = self._get_default_values()["shortcuts"]
        self._load_settings()

    def _reset_all(self):
        ret = QMessageBox.question(self, "Reset Settings", "Are you sure you want to restore all settings to their factory defaults?")
        if ret == QMessageBox.Yes:
            self.settings = self._get_default_values()
            self._load_settings()

if __name__ == "__main__":
    app = QApplication(sys.argv)
    app.setStyle("Fusion")
    
    dialog = SettingsDialog()
    
    # Example signal connection
    dialog.theme_changed.connect(lambda t: print(f"Theme Signal Received: {t}"))
    dialog.accent_color_changed.connect(lambda c: print(f"Accent Signal Received: {c}"))
    
    dialog.show()
    sys.exit(app.exec())
